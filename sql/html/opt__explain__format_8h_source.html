171  struct mem_root_str
172  {
173  const char *str;
174  size_t length;
175  Lazy *deferred; //< encapsulated expression to evaluate it later (on demand)
176 
177  mem_root_str() { cleanup(); }
178  void cleanup()
179  {
180  str= NULL;
181  length= 0;
182  deferred= NULL;
183  }
184  bool is_empty()
185  {
186  if (deferred)
187  {
188  StringBuffer<128> buff(system_charset_info);
189  if (deferred->eval(&buff) || set(buff))
190  {
191  DBUG_ASSERT(!"OOM!");
192  return true; // ignore OOM
193  }
194  deferred= NULL; // prevent double evaluation, if any
195  }
196  return str == NULL;
197  }
198  bool set(const char *str_arg)
199  {
200  return set(str_arg, strlen(str_arg));
201  }
202  bool set(const String &s)
203  {
204  return set(s.ptr(), s.length());
205  }
214  bool set(const char *str_arg, size_t length_arg)
215  {
216  deferred= NULL;
217  if (!(str= strndup_root(current_thd->mem_root, str_arg, length_arg)))
218  return true; /* purecov: inspected */
219  length= length_arg;
220  return false;
221  }
227  void set(Lazy *x)
228  {
229  deferred= x;
230  str= NULL;
231  length= 0;
232  }
239  void set_const(const char *str_arg)
240  {
241  return set_const(str_arg, strlen(str_arg));
242  }
243  void set_const(const char *str_arg, size_t length_arg)
244  {
245  deferred= NULL;
246  str= str_arg;
247  length= length_arg;
248  }
249 
250  static char *strndup_root(MEM_ROOT *root, const char *str, size_t len)
251  {
252  if (len == 0 || str == NULL)
253  return const_cast<char *>("");
254  if (str[len - 1] == 0)
255  return static_cast<char *>(memdup_root(root, str, len));
256 
257  char *ret= static_cast<char*>(alloc_root(root, len + 1));
258  if (ret != NULL)
259  {
260  memcpy(ret, str, len);
261  ret[len]= 0;
262  }
263  return ret;
264  }
265  };

488  {
489  sorts[clause]&= ~property;
490  }
491 
495  bool get(Explain_sort_clause clause, Explain_sort_property property) const
496  {
497  return (sorts[clause] & property) || (sorts[clause] & ESP_CHECKED);
498  }
499 
503  bool any(Explain_sort_property property) const
504  {
505  for (size_t i= ESC_none + 1; i <= ESC_MAX - 1; i++)
506  {

509  }
