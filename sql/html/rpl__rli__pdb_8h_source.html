190  Prealloced_array<Element_type, 1, true> m_Q;
191  ulong size; // the Size of the queue in terms of element
192  ulong avail; // first Available index to append at (next to tail)
193  ulong entry; // the head index or the entry point to the queue.
194  volatile ulong len; // actual length
195  bool inited_queue;
196 
197  circular_buffer_queue(ulong max) :
198  m_Q(PSI_INSTRUMENT_ME),
199  size(max), avail(0), entry(max), len(0), inited_queue(false)
200  {
201  if (!m_Q.reserve(size))
202  inited_queue= true;
203  m_Q.resize(size);
204  }
205  circular_buffer_queue() : m_Q(PSI_INSTRUMENT_ME), inited_queue(false) {}
206  ~circular_buffer_queue ()
207  {
208  }
209 
219  ulong de_queue(Element_type *item);
227  ulong de_tail(Element_type *item);
228 
234  ulong en_queue(Element_type *item);
238  Element_type* head_queue()
239  {
240  if (empty())
241  return NULL;
242  return &m_Q[entry];
243  }
244 
245  bool gt(ulong i, ulong k); // comparision of ordering of two entities
246  /* index is within the valid range */
247  bool in(ulong k) { return !empty() &&
248  (entry > avail ? (k >= entry || k < avail) : (k >= entry && k < avail)); }
249  bool empty() { return entry == size; }
250  bool full() { return avail == size; }
251 };

366 template <typename Element_type>
367 ulong circular_buffer_queue<Element_type>::en_queue(Element_type *item)
368 {
369  ulong ret;
370  if (avail == size)
371  {
372  DBUG_ASSERT(avail == m_Q.size());
373  return (ulong) -1;
374  }
375 
376  // store
377 
378  ret= avail;
379  m_Q[avail]= *item;
380 
381  // pre-boundary cond
382  if (entry == size)
383  entry= avail;
384 
385  avail= (avail + 1) % size;
386  len++;
387 
388  // post-boundary cond
389  if (avail == entry)
390  avail= size;
391 
392  DBUG_ASSERT(avail == entry ||
393  len == (avail >= entry) ?
394  (avail - entry) : (size + avail - entry));
395  DBUG_ASSERT(avail != entry);
396 
397  return ret;
398 }
399

409 template <typename Element_type>
410 ulong circular_buffer_queue<Element_type>::de_queue(Element_type *item)
411 {
412  ulong ret;
413  if (entry == size)
414  {
415  DBUG_ASSERT(len == 0);
416  return (ulong) -1;
417  }
418 
419  ret= entry;
420  *item= m_Q[entry];
421  len--;
422 
423  // pre boundary cond
424  if (avail == size)
425  avail= entry;
426  entry= (entry + 1) % size;
427 
428  // post boundary cond
429  if (avail == entry)
430  entry= size;
431 
432  DBUG_ASSERT(entry == size ||
433  (len == (avail >= entry)? (avail - entry) :
434  (size + avail - entry)));
435  DBUG_ASSERT(avail != entry);
436 
437  return ret;
438 }
439 
440 
441 template <typename Element_type>
442 ulong circular_buffer_queue<Element_type>::de_tail(Element_type *item)
443 {
444  if (entry == size)
445  {
446  DBUG_ASSERT(len == 0);
447  return (ulong) -1;
448  }
449 
450  avail= (entry + len - 1) % size;
451  *item= m_Q[avail];
452  len--;
453 
454  // post boundary cond
455  if (avail == entry)
456  entry= size;
457 
458  DBUG_ASSERT(entry == size ||
459  (len == (avail >= entry)? (avail - entry) :
460  (size + avail - entry)));
461  DBUG_ASSERT(avail != entry);
462 
463  return avail;
464 }

644  void set_rli_description_event(Format_description_log_event *fdle)
645  {
646  DBUG_ENTER("Slave_worker::set_rli_description_event");
647 
648  if (fdle)
649  {
650  /*
651  When the master rotates its binary log, set gtid_next to
652  NOT_YET_DETERMINED. This tells the slave thread that:
653 
654  - If a Gtid_log_event is read subsequently, gtid_next will be
655  set to the given GTID (this is done in
656  gtid_pre_statement_checks()).
657 
658  - If a statement is executed before any Gtid_log_event, then
659  gtid_next is set to anonymous (this is done in
660  Gtid_log_event::do_apply_event().
661 
662  It is imporant to not set GTID_NEXT=NOT_YET_DETERMINED in the
663  middle of a transaction. If that would happen when
664  GTID_MODE=ON, the next statement would fail because it
665  implicitly sets GTID_NEXT=ANONYMOUS, which is disallowed when
666  GTID_MODE=ON. So then there would be no way to end the
667  transaction; any attempt to do so would result in this error.
668  (It is not possible for the slave threads to have
669  gtid_next.type==AUTOMATIC or UNDEFINED in the middle of a
670  transaction, but it is possible for a client thread to have
671  gtid_next.type==AUTOMATIC and issue a BINLOG statement
672  containing this Format_description_log_event.)
673  */
674  if (!is_in_group() &&
675  (info_thd->variables.gtid_next.type == AUTOMATIC_GROUP ||
676  info_thd->variables.gtid_next.type == UNDEFINED_GROUP))
677  {
678  DBUG_PRINT("info", ("Setting gtid_next.type to NOT_YET_DETERMINED_GROUP"));
679  info_thd->variables.gtid_next.set_not_yet_determined();
680  }
681  adapt_to_master_version_updown(fdle->get_product_version(),
682  get_master_server_version());
683  }
684  if (rli_description_event)
685  {
686  DBUG_ASSERT(rli_description_event->usage_counter.atomic_get() > 0);
687 
688  if (rli_description_event->usage_counter.atomic_add(-1) == 1)
689  {
690  /* The being deleted by Worker FD can't be the latest one */
691  DBUG_ASSERT(rli_description_event != c_rli->get_rli_description_event());
692 
693  delete rli_description_event;
694  }
695  }
696  rli_description_event= fdle;
697 
698  DBUG_VOID_RETURN;
699  }
