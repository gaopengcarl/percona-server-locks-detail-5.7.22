272 void Table_cache::link_unused_table(TABLE *table)
273 {
274  if (m_unused_tables)
275  {
276  table->next= m_unused_tables;
277  table->prev= m_unused_tables->prev;
278  m_unused_tables->prev= table;
279  table->prev->next= table;
280  }
281  else
282  m_unused_tables= table->next= table->prev= table;
283  check_unused();
284 }

289 void Table_cache::unlink_unused_table(TABLE *table)
290 {
291  table->next->prev= table->prev;
292  table->prev->next= table->next;
293  if (table == m_unused_tables)
294  {
295  m_unused_tables= m_unused_tables->next;
296  if (table == m_unused_tables)
297  m_unused_tables= NULL;
298  }
299  check_unused();
300 }

312 void Table_cache::free_unused_tables_if_necessary(THD *thd)
313 {
314  /*
315  We have too many TABLE instances around let us try to get rid of them.
316 
317  Note that we might need to free more than one TABLE object, and thus
318  need the below loop, in case when table_cache_size is changed dynamically,
319  at server run time.
320  */
321  if (m_table_count > table_cache_size_per_instance && m_unused_tables)
322  {
323  mysql_mutex_lock(&LOCK_open);
324  while (m_table_count > table_cache_size_per_instance &&
325  m_unused_tables)
326  {
327  TABLE *table_to_free= m_unused_tables;
328  remove_table(table_to_free);
329  intern_close_table(table_to_free);
330  thd->status_var.table_open_cache_overflows++;
331  }
332  mysql_mutex_unlock(&LOCK_open);
333  }
334 }

349 bool Table_cache::add_used_table(THD *thd, TABLE *table)
350 {
351  Table_cache_element *el;
352 
353  assert_owner();
354 
355  DBUG_ASSERT(table->in_use == thd);
356 
357  /*
358  Try to get Table_cache_element representing this table in the cache
359  from array in the TABLE_SHARE.
360  */
361  el= table->s->cache_element[table_cache_manager.cache_index(this)];
362 
363  if (!el)
364  {
365  /*
366  If TABLE_SHARE doesn't have pointer to the element representing table
367  in this cache, the element for the table must be absent from table the
368  cache.
369 
370  Allocate new Table_cache_element object and add it to the cache
371  and array in TABLE_SHARE.
372  */
373  DBUG_ASSERT(! my_hash_search(&m_cache,
374  (uchar*)table->s->table_cache_key.str,
375  table->s->table_cache_key.length));
376 
377  if (!(el= new Table_cache_element(table->s)))
378  return true;
379 
380  if (my_hash_insert(&m_cache, (uchar*)el))
381  {
382  delete el;
383  return true;
384  }
385 
386  table->s->cache_element[table_cache_manager.cache_index(this)]= el;
387  }
388 
389  /* Add table to the used tables list */
390  el->used_tables.push_front(table);
391 
392  m_table_count++;
393 
394  free_unused_tables_if_necessary(thd);
395 
396  return false;
397 }

407 void Table_cache::remove_table(TABLE *table)
408 {
409  Table_cache_element *el=
410  table->s->cache_element[table_cache_manager.cache_index(this)];
411 
412  assert_owner();
413 
414  if (table->in_use)
415  {
416  /* Remove from per-table chain of used TABLE objects. */
417  el->used_tables.remove(table);
418  }
419  else
420  {
421  /* Remove from per-table chain of unused TABLE objects. */
422  el->free_tables.remove(table);
423 
424  /* And per-cache unused chain. */
425  unlink_unused_table(table);
426  }
427 
428  m_table_count--;
429 
430  if (el->used_tables.is_empty() && el->free_tables.is_empty())
431  {
432  (void) my_hash_delete(&m_cache, (uchar*) el);
433  /*
434  Remove reference to deleted cache element from array
435  in the TABLE_SHARE.
436  */
437  table->s->cache_element[table_cache_manager.cache_index(this)]= NULL;
438  }
439 }

464 TABLE* Table_cache::get_table(THD *thd, my_hash_value_type hash_value,
465  const char *key, size_t key_length,
466  TABLE_SHARE **share)
467 {
468  Table_cache_element *el;
469  TABLE *table;
470 
471  assert_owner();
472 
473  *share= NULL;
474 
475  if (!(el= (Table_cache_element*) my_hash_search_using_hash_value(&m_cache,
476  hash_value, (uchar*) key, key_length)))
477  return NULL;
478 
479  *share= el->share;
480 
481  if ((table= el->free_tables.front()))
482  {
483  DBUG_ASSERT(!table->in_use);
484 
485  /*
486  Unlink table from list of unused TABLE objects for this
487  table in this cache.
488  */
489  el->free_tables.remove(table);
490 
491  /* Unlink table from unused tables list for this cache. */
492  unlink_unused_table(table);
493 
494  /*
495  Add table to list of used TABLE objects for this table
496  in the table cache.
497  */
498  el->used_tables.push_front(table);
499 
500  table->in_use= thd;
501  /* The ex-unused table must be fully functional. */
502  DBUG_ASSERT(table->db_stat && table->file);
503  /* The children must be detached from the table. */
504  DBUG_ASSERT(! table->file->extra(HA_EXTRA_IS_ATTACHED_CHILDREN));
505  }
506 
507  return table;
508 }

519 void Table_cache::release_table(THD *thd, TABLE *table)
520 {
521  Table_cache_element *el=
522  table->s->cache_element[table_cache_manager.cache_index(this)];
523 
524  assert_owner();
525 
526  DBUG_ASSERT(table->in_use);
527  DBUG_ASSERT(table->file);
528 
529  /* We shouldn't put the table to 'unused' list if the share is old. */
530  DBUG_ASSERT(! table->s->has_old_version());
531 
532  table->in_use= NULL;
533 
534  /* Remove TABLE from the list of used objects for the table in this cache. */
535  el->used_tables.remove(table);
536  /* Add TABLE to the list of unused objects for the table in this cache. */
537  el->free_tables.push_front(table);
538  /* Also link it last in the list of unused TABLE objects for the cache. */
539  link_unused_table(table);
540 
541  /*
542  We free the least used tables, not the subject table, to keep the LRU order.
543  Note that in most common case the below call won't free anything.
544  */
545  free_unused_tables_if_necessary(thd);
546 }
