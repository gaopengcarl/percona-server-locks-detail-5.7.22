19 /*
20  Such interval is "discrete": it is the set of
21  { auto_inc_interval_min + k * increment,
22  0 <= k <= (auto_inc_interval_values-1) }
23  Where "increment" is maintained separately by the user of this class (and is
24  currently only thd->variables.auto_increment_increment).
25  It mustn't derive from Sql_alloc, because SET INSERT_ID needs to
26  allocate memory which must stay allocated for use by the next statement.
27 */
28 class Discrete_interval {
29 private:
30  ulonglong interval_min;
31  ulonglong interval_values;
32  ulonglong interval_max; // excluded bound. Redundant.
33 public:
34  Discrete_interval *next; // used when linked into Discrete_intervals_list
35 
37  bool in_range(const ulonglong value) const
38  {
39  return ((value >= interval_min) && (value < interval_max));
40  }
41 
42  void replace(ulonglong start, ulonglong val, ulonglong incr)
43  {
44  interval_min= start;
45  interval_values= val;
46  interval_max= (val == ULLONG_MAX) ? val : start + val * incr;
47  }
48  Discrete_interval(ulonglong start, ulonglong val, ulonglong incr) :
49  next(NULL) { replace(start, val, incr); };
50  Discrete_interval() : next(NULL) { replace(0, 0, 0); };
51  ulonglong minimum() const { return interval_min; };
52  ulonglong values() const { return interval_values; };
53  ulonglong maximum() const { return interval_max; };
54  /*
55  If appending [3,5] to [1,2], we merge both in [1,5] (they should have the
56  same increment for that, user of the class has to ensure that). That is
57  just a space optimization. Returns 0 if merge succeeded.
58  */
59  bool merge_if_contiguous(ulonglong start, ulonglong val, ulonglong incr)
60  {
61  if (interval_max == start)
62  {
63  if (val == ULLONG_MAX)
64  {
65  interval_values= interval_max= val;
66  }
67  else
68  {
69  interval_values+= val;
70  interval_max= start + val * incr;
71  }
72  return 0;
73  }
74  return 1;
75  };
76 };

181  {
182  // If there are no intervals, add one.
183  if (head == NULL)
184  {
