94 /*
95  This class is the construction element for serialization of Item tree
96  in condition pushdown.
97  An instance of Ndb_Item represents a constant, table field reference,
98  unary or binary comparison predicate, and start/end of AND/OR.
99  Instances of Ndb_Item are stored in a linked list implemented by Ndb_cond
100  class.
101  The order of elements produced by Ndb_cond::next corresponds to
102  breadth-first traversal of the Item (i.e. expression) tree in prefix order.
103  AND and OR have arbitrary arity, so the end of AND/OR group is marked with
104  Ndb_item with type == NDB_END_COND.
105  NOT items represent negated conditions and generate NAND/NOR groups.
106 */
107 class Ndb_item : public Sql_alloc
108 {
109 public:
110  Ndb_item(NDB_ITEM_TYPE item_type) : type(item_type) {}
111  Ndb_item(NDB_ITEM_TYPE item_type,
112  NDB_ITEM_QUALIFICATION item_qualification,
113  const Item *item_value)
114  : type(item_type), qualification(item_qualification)
115  {
116  switch(item_type) {
117  case(NDB_VALUE):
118  value.item= item_value;
119  break;
120  case(NDB_FIELD): {
121  NDB_ITEM_FIELD_VALUE *field_value= new NDB_ITEM_FIELD_VALUE();
122  Item_field *field_item= (Item_field *) item_value;
123  field_value->field= field_item->field;
124  field_value->column_no= -1; // Will be fetched at scan filter generation
125  value.field_value= field_value;
126  break;
127  }
128  case(NDB_FUNCTION):
129  value.item= item_value;
130  value.arg_count= ((Item_func *) item_value)->argument_count();
131  break;
132  case(NDB_END_COND):
133  break;
134  }
135  }
136  Ndb_item(Field *field, int column_no) : type(NDB_FIELD)
137  {
138  NDB_ITEM_FIELD_VALUE *field_value= new NDB_ITEM_FIELD_VALUE();
139  qualification.field_type= field->real_type();
140  field_value->field= field;
141  field_value->column_no= column_no;
142  value.field_value= field_value;
143  }
144  Ndb_item(Item_func::Functype func_type, const Item *item_value)
145  : type(NDB_FUNCTION)
146  {
147  qualification.function_type= item_func_to_ndb_func(func_type);
148  value.item= item_value;
149  value.arg_count= ((Item_func *) item_value)->argument_count();
150  }
151  Ndb_item(Item_func::Functype func_type, uint no_args)
152  : type(NDB_FUNCTION)
153  {
154  qualification.function_type= item_func_to_ndb_func(func_type);
155  value.arg_count= no_args;
156  }
157  ~Ndb_item()
158  {
159  if (type == NDB_FIELD)
160  {
161  delete value.field_value;
162  value.field_value= NULL;
163  }
164  }
165 
166  uint32 pack_length()
167  {
168  switch(type) {
169  case(NDB_VALUE):
170  if(qualification.value_type == Item::STRING_ITEM)
171  return value.item->str_value.length();
172  break;
173  case(NDB_FIELD):
174  return value.field_value->field->pack_length();
175  default:
176  break;
177  }
178 
179  return 0;
180  }
181 
182  Field * get_field() { return value.field_value->field; }
183 
184  int get_field_no() { return value.field_value->column_no; }
185 
186  int argument_count()
187  {
188  return value.arg_count;
189  }
190 
191  const char* get_val()
192  {
193  switch(type) {
194  case(NDB_VALUE):
195  if(qualification.value_type == Item::STRING_ITEM)
196  return value.item->str_value.ptr();
197  break;
198  case(NDB_FIELD):
199  return (char*) value.field_value->field->ptr;
200  case(NDB_FUNCTION):
201  if(qualification.value_type == Item::STRING_ITEM)
202  return value.item->str_value.ptr();
203  default:
204  break;
205  }
206 
207  return NULL;
208  }

249  uint32 save_in_field(Ndb_item *field_item)
250  {
251  uint32 length= 0;
252  DBUG_ENTER("save_in_field");
253  Field *field = field_item->value.field_value->field;
254  const Item *item= value.item;
255  if (item && field)
256  {
257  length= item->max_length;
258  my_bitmap_map *old_map=
259  dbug_tmp_use_all_columns(field->table, field->table->write_set);
260  ((Item *)item)->save_in_field(field, false);
261  dbug_tmp_restore_column_map(field->table->write_set, old_map);
262  }
263  DBUG_RETURN(length);
264  }

266  static NDB_FUNC_TYPE item_func_to_ndb_func(Item_func::Functype fun)
267  {
268  switch (fun) {
269  case (Item_func::EQ_FUNC): { return NDB_EQ_FUNC; }
270  case (Item_func::NE_FUNC): { return NDB_NE_FUNC; }
271  case (Item_func::LT_FUNC): { return NDB_LT_FUNC; }
272  case (Item_func::LE_FUNC): { return NDB_LE_FUNC; }
273  case (Item_func::GT_FUNC): { return NDB_GT_FUNC; }
274  case (Item_func::GE_FUNC): { return NDB_GE_FUNC; }
275  case (Item_func::ISNULL_FUNC): { return NDB_ISNULL_FUNC; }
276  case (Item_func::ISNOTNULL_FUNC): { return NDB_ISNOTNULL_FUNC; }
277  case (Item_func::LIKE_FUNC): { return NDB_LIKE_FUNC; }
278  case (Item_func::NOT_FUNC): { return NDB_NOT_FUNC; }
279  case (Item_func::NEG_FUNC): { return NDB_UNKNOWN_FUNC; }
280  case (Item_func::UNKNOWN_FUNC): { return NDB_UNKNOWN_FUNC; }
281  case (Item_func::COND_AND_FUNC): { return NDB_COND_AND_FUNC; }
282  case (Item_func::COND_OR_FUNC): { return NDB_COND_OR_FUNC; }
283  default: { return NDB_UNSUPPORTED_FUNC; }
284  }
285  }
286 
287  static NDB_FUNC_TYPE negate(NDB_FUNC_TYPE fun)
288  {
289  uint i= (uint) fun;
290  DBUG_ASSERT(fun == neg_map[i].pos_fun);
291  return neg_map[i].neg_fun;
292  }

294  NDB_ITEM_TYPE type;
295  NDB_ITEM_QUALIFICATION qualification;
296  private:
297  NDB_ITEM_VALUE value;

300 /*
301  This class implements a linked list used for storing a
302  serialization of the Item tree for condition pushdown.
303  */
304 class Ndb_cond : public Sql_alloc
305 {
306  public:
307  Ndb_cond() : ndb_item(NULL), next(NULL), prev(NULL) {}
308  ~Ndb_cond()
309  {
310  if (ndb_item) delete ndb_item;
311  ndb_item= NULL;
312  /*
313  First item in the linked list deletes all in a loop
314  Note - doing it recursively causes stack issues for
315  big IN clauses
316  */
317  Ndb_cond *n= next;
318  while (n)
319  {
320  Ndb_cond *tmp= n;
321  n= n->next;
322  tmp->next= NULL;
323  delete tmp;
324  }
325  next= prev= NULL;
326  }
327  Ndb_item *ndb_item;
328  Ndb_cond *next;
329  Ndb_cond *prev;
330 };

354 /*
355  This class implements look-ahead during the parsing
356  of the item tree. It contains bit masks for expected
357  items, field types and field results. It also contains
358  expected collation. The parse context (Ndb_cond_traverse_context)
359  always contains one expect_stack instance (top of the stack).
360  More expects (deeper look-ahead) can be pushed to the expect_stack
361  to check specific order (currently used for detecting support for
362  <field> LIKE <string>|<func>, but not <string>|<func> LIKE <field>).
363  */
364 class Ndb_expect_stack : public Sql_alloc
365 {
366  static const uint MAX_EXPECT_ITEMS = Item::VIEW_FIXER_ITEM + 1;
367  static const uint MAX_EXPECT_FIELD_TYPES = MYSQL_TYPE_GEOMETRY + 1;
368  static const uint MAX_EXPECT_FIELD_RESULTS = DECIMAL_RESULT + 1;
369  public:
370 Ndb_expect_stack(): collation(NULL), length(0), max_length(0), next(NULL)
371  {
372  // Allocate type checking bitmaps using fixed size buffers
373  // since max size is known at compile time
374  bitmap_init(&expect_mask, m_expect_buf,
375  MAX_EXPECT_ITEMS, FALSE);
376  bitmap_init(&expect_field_type_mask, m_expect_field_type_buf,
377  MAX_EXPECT_FIELD_TYPES, FALSE);
378  bitmap_init(&expect_field_result_mask, m_expect_field_result_buf,
379  MAX_EXPECT_FIELD_RESULTS, FALSE);
380  }
381  ~Ndb_expect_stack()
382  {
383  if (next)
384  delete next;
385  next= NULL;
386  }
387  void push(Ndb_expect_stack* expect_next)
388  {
389  next= expect_next;
390  }
391  void pop()
392  {
393  if (next)
394  {
395  Ndb_expect_stack* expect_next= next;
396  bitmap_clear_all(&expect_mask);
397  bitmap_union(&expect_mask, &next->expect_mask);
398  bitmap_clear_all(&expect_field_type_mask);
399  bitmap_union(&expect_field_type_mask, &next->expect_field_type_mask);
400  bitmap_clear_all(&expect_field_result_mask);
401  bitmap_union(&expect_field_result_mask, &next->expect_field_result_mask);
402  collation= next->collation;
403  next= next->next;
404  delete expect_next;
405  }
406  }
407  void expect(Item::Type type)
408  {
409  bitmap_set_bit(&expect_mask, (uint) type);
410  if (type == Item::FIELD_ITEM)
411  expect_all_field_types();
412  }
413  void dont_expect(Item::Type type)
414  {
415  bitmap_clear_bit(&expect_mask, (uint) type);
416  }
417  bool expecting(Item::Type type)
418  {
419  if (unlikely((uint)type > MAX_EXPECT_ITEMS))
420  {
421  // Unknown type, can't be expected
422  return false;
423  }
424  return bitmap_is_set(&expect_mask, (uint) type);
425  }
426  void expect_nothing()
427  {
428  bitmap_clear_all(&expect_mask);
429  }
430  bool expecting_nothing()
431  {
432  return bitmap_is_clear_all(&expect_mask);
433  }
434  void expect_only(Item::Type type)
435  {
436  expect_nothing();
437  expect(type);
438  }
439 
440  void expect_field_type(enum_field_types type)
441  {
442  bitmap_set_bit(&expect_field_type_mask, (uint) type);
443  }
444  void expect_all_field_types()
445  {
446  bitmap_set_all(&expect_field_type_mask);
447  }
448  bool expecting_field_type(enum_field_types type)
449  {
450  if (unlikely((uint)type > MAX_EXPECT_FIELD_TYPES))
451  {
452  // Unknown type, can't be expected
453  return false;
454  }
455  return bitmap_is_set(&expect_field_type_mask, (uint) type);
456  }
457  void expect_no_field_type()
458  {
459  bitmap_clear_all(&expect_field_type_mask);
460  }
461  bool expecting_no_field_type()
462  {
463  return bitmap_is_clear_all(&expect_field_type_mask);
464  }
465  void expect_only_field_type(enum_field_types result)
466  {
467  expect_no_field_type();

469  }
470 
471  void expect_field_result(Item_result result)
472  {
473  bitmap_set_bit(&expect_field_result_mask, (uint) result);
474  }
475  bool expecting_field_result(Item_result result)
476  {
477  if (unlikely((uint)result > MAX_EXPECT_FIELD_RESULTS))
478  {
479  // Unknown result, can't be expected
480  return false;
481  }
482  return bitmap_is_set(&expect_field_result_mask,
483  (uint) result);
484  }
485  void expect_no_field_result()
486  {
487  bitmap_clear_all(&expect_field_result_mask);
488  }
489  bool expecting_no_field_result()
490  {
491  return bitmap_is_clear_all(&expect_field_result_mask);
492  }
493  void expect_only_field_result(Item_result result)
494  {
495  expect_no_field_result();

497  }
498  void expect_collation(const CHARSET_INFO* col)

501  }

511  void expect_length(Uint32 len)

518  }
