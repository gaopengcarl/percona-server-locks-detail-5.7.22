38 inline char *serialize_xid(char *buf, long fmt, long gln, long bln,
39  const char *dat)
40 {
41  int i;
42  char *c= buf;
43  /*
44  Build a string like following pattern:
45  X'hex11hex12...hex1m',X'hex21hex22...hex2n',11
46  and store it into buf.
47  Here hex1i and hex2k are hexadecimals representing XID's internal
48  raw bytes (1 <= i <= m, 1 <= k <= n), and `m' and `n' even numbers
49  half of which corresponding to the lengths of XID's components.
50  */
51  *c++= 'X';
52  *c++= '\'';
53  for (i= 0; i < gln; i++)
54  {
55  *c++=_dig_vec_lower[((uchar*) dat)[i] >> 4];
56  *c++=_dig_vec_lower[((uchar*) dat)[i] & 0x0f];
57  }
58  *c++= '\'';
59 
60  *c++= ',';
61  *c++= 'X';
62  *c++= '\'';
63  for (; i < gln + bln; i++)
64  {
65  *c++=_dig_vec_lower[((uchar*) dat)[i] >> 4];
66  *c++=_dig_vec_lower[((uchar*) dat)[i] & 0x0f];
67  }
68  *c++= '\'';
69  sprintf(c, ",%lu", fmt);
70 
71  return buf;
72 }
