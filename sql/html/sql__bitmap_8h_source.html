54  // Use a spearate temporary buffer, as bitmap_init() clears all the bits.
55  ulonglong buf2;
56  MY_BITMAP map2;

58  bitmap_init(&map2, (uint32 *) &buf2, sizeof(ulonglong) * 8, 0);
59 
60  // Store the original bits.

62  int8store(const_cast<uchar *>(static_cast<uchar *>

67  int4store(const_cast<uchar *>(static_cast<uchar *>

70  }

73  }

171 /* An iterator to quickly walk over bits in unlonglong bitmap. */
172 class Table_map_iterator
173 {
174  ulonglong bmp;
175  uint no;
176 public:
177  Table_map_iterator(ulonglong t) : bmp(t), no(0) {}
178  int next_bit()
179  {
180  static const char last_bit[16]= {32, 0, 1, 0,
181  2, 0, 1, 0,
182  3, 0, 1, 0,
183  2, 0, 1, 0};
184  uint bit;
185  while ((bit= last_bit[bmp & 0xF]) == 32)
186  {
187  no += 4;
188  bmp= bmp >> 4;
189  if (!bmp)
190  return BITMAP_END;
191  }
192  bmp &= ~(1LL << bit);
193  return no + bit;
194  }
195  enum { BITMAP_END= 64 };
196 };
