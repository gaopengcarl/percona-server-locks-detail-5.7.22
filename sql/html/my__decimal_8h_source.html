121  {
122  rhs.sanity_check();
123 #if !defined(DBUG_OFF)
124  foo1= test_value;
125  foo2= test_value;
126 #endif

224 inline int check_result_and_overflow(uint mask, int result, my_decimal *val)
225 {
226  if (val->check_result(mask, result) & E_DEC_OVERFLOW)
227  {
228  bool sign= val->sign();
229  val->sanity_check();
230  max_internal_decimal(val);
231  val->sign(sign);
232  }
233  /*
234  Avoid returning negative zero, cfr. decimal_cmp()
235  For result == E_DEC_DIV_ZERO *val has not been assigned.
236  */
237  if (result != E_DEC_DIV_ZERO && val->sign() && decimal_is_zero(val))
238  val->sign(false);
239  return result;
240 }

242 inline uint my_decimal_length_to_precision(uint length, uint scale,
243  bool unsigned_flag)
244 {
245  /* Precision can't be negative thus ignore unsigned_flag when length is 0. */
246  DBUG_ASSERT(length || !scale);

252 inline uint32 my_decimal_precision_to_length_no_truncation(uint precision,
253  uint8 scale,
254  bool unsigned_flag)
255 {

257  When precision is 0 it means that original length was also 0. Thus
258  unsigned_flag is ignored in this case.
